    <script>
    const canvas = document.getElementById('gameCanvas');
    const ctx = canvas.getContext('2d');
    const W = canvas.width, H = canvas.height;
    const WALL = 25;

    let gameState = 'countdown';
    let gameTime = 0;
    let lastTime = performance.now();
    let screenShake = { x: 0, y: 0, intensity: 0 };
    let totalDamage = { A: 0, M: 0, B: 0 };
    let gameEndTimeout = null;

    const particles = [];
    const bgParticles = [];
    const damageNumbers = [];
    const poisonPools = [];
    const projectiles = [];
    const explosions = [];

    for (let i = 0; i < 60; i++) {
        bgParticles.push({
            x: Math.random() * W, y: Math.random() * H,
            vx: (Math.random() - 0.5) * 0.3, vy: (Math.random() - 0.5) * 0.3,
            size: Math.random() * 2 + 0.5, alpha: Math.random() * 0.3 + 0.1
        });
    }

    const keys = {};
    document.addEventListener('keydown', e => { keys[e.key.toLowerCase()] = true; if (e.key.toLowerCase() === 'r') restart(); });
    document.addEventListener('keyup', e => keys[e.key.toLowerCase()] = false);

    class Ball {
        constructor(x, y, radius, color, type) {
            this.x = x; this.y = y; this.radius = radius;
            this.color = color; this.type = type;
            this.vx = 0; this.vy = 0;
            this.hp = 100; this.maxHp = 100;
            this.speed = 2.5; this.friction = 0.992;
            this.hitFlash = 0; this.invincible = 0;

            // Sword
            this.swordAngle = 0; this.swordLength = radius * 3.2;
            this.swordWidth = 7; this.swordRotSpeed = 3;
            this.isSticking = false; this.stickTarget = null;
            this.stickHits = 0; this.stickTimer = 0;

            // Poison
            this.poisonDOT = 0; this.poisonTimer = 0;
            this.poisonDuration = 5; this.poisonDamage = 6; // Buffed from 4 to 6
            this.poisonAura = 0; this.auraPhase = 0;

            // Mage
            this.fireCooldown = 0; this.fireRate = 1.8; // Slower firing
            this.projectileSpeed = 4; this.projectileDamage = 10; // Less damage
            this.freezeDuration = 0.8; // Shortened from 2s to 0.8s
            this.freezeImmunity = 0; // Can't be frozen again during immunity
            this.isFrozen = 0;
            this.mageAngle = 0;

            // AI
            this.aiTimer = 0; this.stuckTimer = 0;
            this.lastPos = { x, y };
        }

        update(dt, others) {
            if (this.isFrozen > 0) {
                this.isFrozen -= dt;
                this.freezeImmunity = 0.5; // Immunity lasts 0.5s after thaw
                this.vx *= 0.85; this.vy *= 0.85;
                if (this.isFrozen <= 0) { this.isFrozen = 0; }
            } else {
                this.freezeImmunity = Math.max(0, this.freezeImmunity - dt);
                this.invincible = Math.max(0, this.invincible - dt);
                this.auraPhase += dt * 3;

                if (gameState === 'playing') this.updateAI(others, dt);

                this.vx *= this.friction;
                this.vy *= this.friction;

                const spd = Math.sqrt(this.vx * this.vx + this.vy * this.vy);
                if (spd > this.speed) { this.vx = (this.vx / spd) * this.speed; this.vy = (this.vy / spd) * this.speed; }
                this.x += this.vx; this.y += this.vy;
            }

            // Wall collision (no poison on wall for poison ball - only on collision with other balls)
            if (this.x - this.radius < WALL) { this.x = WALL + this.radius; this.vx *= -0.95; }
            if (this.x + this.radius > W - WALL) { this.x = W - WALL - this.radius; this.vx *= -0.95; }
            if (this.y - this.radius < WALL) { this.y = WALL + this.radius; this.vy *= -0.95; }
            if (this.y + this.radius > H - WALL) { this.y = H - WALL - this.radius; this.vy *= -0.95; }

            // Sword rotation
            if (this.type === 'A') {
                this.swordAngle += this.swordRotSpeed * dt;
                const tipX = this.x + Math.cos(this.swordAngle) * this.swordLength;
                const tipY = this.y + Math.sin(this.swordAngle) * this.swordLength;
                swordTrail.push({ x: tipX, y: tipY, alpha: 1, color: '#ff6b6b' });
                if (swordTrail.length > 12) swordTrail.shift();
            }

            // Mage aim
            if (this.type === 'M') {
                const nearest = this.findNearest(others);
                if (nearest) {
                    const dx = nearest.x - this.x, dy = nearest.y - this.y;
                    this.mageAngle = Math.atan2(dy, dx);
                }
                this.fireCooldown -= dt;
                if (this.fireCooldown <= 0 && gameState === 'playing' && !this.isFrozen) {
                    this.fireProjectile(others);
                    this.fireCooldown = this.fireRate;
                }
            }

            if (this.hitFlash > 0) this.hitFlash -= dt * 4;

            if (this.poisonDOT > 0) {
                this.hp -= this.poisonDamage * dt;
                this.poisonTimer -= dt;
                this.poisonAura = Math.sin(this.auraPhase) * 0.3 + 0.7;
                if (this.poisonTimer <= 0) { this.poisonDOT = 0; this.poisonAura = 0; }
            }

            // Death check - mark as dead but don't end game immediately
            if (this.hp <= 0 && this.hp > -999) {
                this.hp = -1000; // Mark as dead
                spawnParticles(this.x, this.y, this.color, 30);
                if (this.type === 'B') createPoisonPool(this.x, this.y);
                checkWinner();
            }
        }

        findNearest(others) {
            let nearest = null, minDist = Infinity;
            for (const o of others) {
                if (o === this || o.hp <= 0) continue;
                const dx = o.x - this.x, dy = o.y - this.y;
                const dist = Math.sqrt(dx * dx + dy * dy);
                if (dist < minDist) { minDist = dist; nearest = o; }
            }
            return nearest;
        }

        updateAI(others, dt) {
            this.aiTimer -= dt;
            if (this.aiTimer > 0) return;
            this.aiTimer = 0.05 + Math.random() * 0.1;

            const nearest = this.findNearest(others);
            if (!nearest) return;

            const dx = nearest.x - this.x, dy = nearest.y - this.y;
            const dist = Math.sqrt(dx * dx + dy * dy);

            const moved = Math.abs(this.x - this.lastPos.x) + Math.abs(this.y - this.lastPos.y);
            if (moved < 0.5) { this.stuckTimer += dt; if (this.stuckTimer > 0.5) { this.vx += (Math.random() - 0.5) * 3; this.vy += (Math.random() - 0.5) * 3; this.stuckTimer = 0; } }
            else this.stuckTimer = 0;
            this.lastPos.x = this.x; this.lastPos.y = this.y;

            let avoidX = 0, avoidY = 0;
            // Only non-poison balls avoid pools
            if (this.type !== 'B') {
                for (const pool of poisonPools) {
                    const pdx = this.x - pool.x, pdy = this.y - pool.y;
                    const pdist = Math.sqrt(pdx * pdx + pdy * pdy);
                    if (pdist < pool.radius * 2.5) { avoidX += (pdx / pdist) * 3; avoidY += (pdy / pdist) * 3; }
                }
            }

            if (this.type === 'A') {
                const idealDist = this.swordLength * 1.2;
                if (dist > idealDist) { const a = Math.atan2(dy, dx); this.vx += Math.cos(a) * 0.5; this.vy += Math.sin(a) * 0.5; }
                else if (dist < idealDist * 0.7) { const a = Math.atan2(dy, dx); this.vx -= Math.cos(a) * 0.3; this.vy -= Math.sin(a) * 0.3; }
                else { const a = Math.atan2(dy, dx) + Math.PI / 2; this.vx += Math.cos(a) * 0.35 * (Math.random() > 0.5 ? 1 : -1); this.vy += Math.sin(a) * 0.35 * (Math.random() > 0.5 ? 1 : -1); }
            } else if (this.type === 'M') {
                const idealDist = 180;
                if (dist < idealDist * 0.6) { const a = Math.atan2(-dy, -dx); this.vx += Math.cos(a) * 0.6; this.vy += Math.sin(a) * 0.6; }
                else if (dist > idealDist * 1.8) { const a = Math.atan2(dy, dx); this.vx += Math.cos(a) * 0.4; this.vy += Math.sin(a) * 0.4; }
                else { const a = Math.atan2(dy, dx) + Math.PI / 2.5 * (Math.random() > 0.5 ? 1 : -1); this.vx += Math.cos(a) * 0.25; this.vy += Math.sin(a) * 0.25; }
            } else {
                // Poison ball - AGGRESSIVE (immune to pools, creates clouds on collision)
                if (this.poisonDOT > 0) { const a = Math.atan2(-dy, -dx); this.vx += Math.cos(a) * 0.6; this.vy += Math.sin(a) * 0.6; }
                else if (dist > this.radius * 3) {
                    // Chase aggressively when far
                    const a = Math.atan2(dy, dx); this.vx += Math.cos(a) * 0.6; this.vy += Math.sin(a) * 0.6;
                } else {
                    // Circle and deal damage when close
                    const a = Math.atan2(dy, dx) + Math.PI / 3; this.vx += Math.cos(a) * 0.5; this.vy += Math.sin(a) * 0.5;
                }

                // Poison ball is immune to pools, gains speed boost
                if (this.speed < 3.0) this.speed = Math.min(3.0, this.speed + 0.01);
            }

            this.vx += avoidX * 0.4; this.vy += avoidY * 0.4;
        }

        fireProjectile(others) {
            const nearest = this.findNearest(others);
            if (!nearest) return;

            const dx = nearest.x - this.x, dy = nearest.y - this.y;
            const angle = Math.atan2(dy, dx);

            projectiles.push({
                x: this.x + Math.cos(angle) * (this.radius + 10),
                y: this.y + Math.sin(angle) * (this.radius + 10),
                vx: Math.cos(angle) * this.projectileSpeed,
                vy: Math.sin(angle) * this.projectileSpeed,
                radius: 8,
                damage: this.projectileDamage,
                owner: this.type,
                trail: []
            });
            spawnParticles(
                this.x + Math.cos(angle) * (this.radius + 10),
                this.y + Math.sin(angle) * (this.radius + 10),
                '#8b5cf6', 6
            );
        }

        takeDamage(amount, fromX, fromY) {
            if (this.invincible > 0) return;
            this.hp -= amount;
            this.hitFlash = 1; this.invincible = 0.05;
            spawnDamageNumber(this.x, this.y - this.radius - 15, Math.round(amount));
            const angle = Math.atan2(this.y - fromY, this.x - fromX);
            spawnParticles(this.x, this.y, this.color, 8, angle);
            triggerScreenShake(amount * 0.25);
        }

        applyFreeze(duration) {
            if (this.freezeImmunity > 0) return; // Immune!
            if (this.isFrozen > 0) return; // Already frozen

            this.isFrozen = duration;
            spawnParticles(this.x, this.y, '#00d9ff', 15);
            spawnDamageNumber(this.x, this.y - this.radius - 30, 'ICE!');
        }

        draw() {
            if (this.hp <= -999) return; // Dead

            ctx.save();

            if (this.poisonDOT > 0) {
                ctx.beginPath(); ctx.arc(this.x, this.y, this.radius + 15, 0, Math.PI * 2);
                ctx.fillStyle = `rgba(189, 147, 249, ${this.poisonAura * 0.4})`; ctx.fill();
            }

            if (this.isFrozen > 0) {
                ctx.beginPath(); ctx.arc(this.x, this.y, this.radius + 12, 0, Math.PI * 2);
                ctx.fillStyle = `rgba(0, 217, 255, ${0.4 + Math.sin(performance.now() / 80) * 0.2})`; ctx.fill();

                ctx.save(); ctx.translate(this.x, this.y);
                for (let i = 0; i < 6; i++) {
                    ctx.rotate(Math.PI / 3);
                    ctx.fillStyle = `rgba(0, 217, 255, ${0.5 + Math.sin(performance.now() / 100 + i) * 0.3})`;
                    ctx.fillRect(-2, this.radius + 5, 4, 10);
                }
                ctx.restore();
            }

            const glowSize = this.radius * (this.isSticking ? 2.5 : 1.8);
            const glow = ctx.createRadialGradient(this.x, this.y, 0, this.x, this.y, glowSize);
            glow.addColorStop(0, this.color + '50'); glow.addColorStop(0.5, this.color + '20'); glow.addColorStop(1, 'transparent');
            ctx.beginPath(); ctx.arc(this.x, this.y, glowSize, 0, Math.PI * 2); ctx.fillStyle = glow; ctx.fill();

            if (this.isSticking && this.stickTarget && this.stickTarget.hp > -999) {
                ctx.beginPath(); ctx.arc(this.stickTarget.x, this.stickTarget.y, this.stickTarget.radius + 8, 0, Math.PI * 2);
                ctx.strokeStyle = '#ff6b6b'; ctx.lineWidth = 3; ctx.setLineDash([8, 4]); ctx.stroke(); ctx.setLineDash([]);
            }

            ctx.beginPath(); ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2);
            ctx.fillStyle = this.hitFlash > 0 ? `rgba(255,255,255,${this.hitFlash})` : (this.invincible > 0 ? this.color + 'aa' : this.color);
            ctx.fill();

            ctx.beginPath(); ctx.arc(this.x - this.radius * 0.3, this.y - this.radius * 0.3, this.radius * 0.35, 0, Math.PI * 2);
            ctx.fillStyle = 'rgba(255,255,255,0.35)'; ctx.fill();

            ctx.beginPath(); ctx.arc(this.x, this.y, this.radius - 2, 0, Math.PI * 2);
            ctx.strokeStyle = 'rgba(255,255,255,0.2)'; ctx.lineWidth = 2; ctx.stroke();

            if (this.type === 'A') {
                ctx.save(); ctx.translate(this.x, this.y); ctx.rotate(this.swordAngle);
                ctx.shadowColor = '#ffffff'; ctx.shadowBlur = 15;
                ctx.fillStyle = '#e8e8e8';
                ctx.fillRect(this.radius * 0.6, -this.swordWidth / 2, this.swordLength - this.radius * 0.5, this.swordWidth);
                ctx.fillStyle = '#ffffff'; ctx.fillRect(this.radius * 0.6, -this.swordWidth / 2, this.swordLength - this.radius * 0.5, 2);
                ctx.beginPath(); ctx.moveTo(this.swordLength + 5, 0); ctx.lineTo(this.swordLength - 10, -10); ctx.lineTo(this.swordLength - 10, 10); ctx.closePath(); ctx.fillStyle = '#e8e8e8'; ctx.fill();
                ctx.fillStyle = '#ffd700'; ctx.fillRect(this.radius * 0.4, -this.swordWidth - 3, 6, this.swordWidth * 2 + 6);
                ctx.restore();
            }

            if (this.type === 'M') {
                ctx.save(); ctx.translate(this.x, this.y); ctx.rotate(this.mageAngle);
                ctx.strokeStyle = '#c4b5fd'; ctx.lineWidth = 3; ctx.shadowColor = '#8b5cf6'; ctx.shadowBlur = 10;
                ctx.beginPath(); ctx.moveTo(this.radius * 0.5, 0); ctx.lineTo(this.radius + 15, 0); ctx.stroke();
                ctx.fillStyle = '#8b5cf6'; ctx.beginPath(); ctx.arc(this.radius + 15, 0, 6, 0, Math.PI * 2); ctx.fill();
                ctx.restore();

                ctx.fillStyle = '#fff';
                ctx.beginPath(); ctx.arc(this.x - 5, this.y - 4, 4, 0, Math.PI * 2); ctx.fill();
                ctx.beginPath(); ctx.arc(this.x + 5, this.y - 4, 4, 0, Math.PI * 2); ctx.fill();
                ctx.fillStyle = '#1a1a2e';
                ctx.beginPath(); ctx.arc(this.x - 4, this.y - 4, 2, 0, Math.PI * 2); ctx.fill();
                ctx.beginPath(); ctx.arc(this.x + 6, this.y - 4, 2, 0, Math.PI * 2); ctx.fill();
            }

            if (this.type === 'B') {
                // Poison bubbles
                for (let i = 0; i < 3; i++) {
                    const bx = this.x + Math.sin(performance.now() / 200 + i * 2) * 10;
                    const by = this.y + Math.cos(performance.now() / 200 + i * 2) * 10;
                    ctx.beginPath(); ctx.arc(bx, by, 3, 0, Math.PI * 2);
                    ctx.fillStyle = `rgba(189, 147, 249, ${0.3 + Math.sin(performance.now() / 100 + i) * 0.2})`; ctx.fill();
                }
            }

            ctx.font = 'bold 10px sans-serif'; ctx.textAlign = 'center';
            ctx.fillStyle = 'rgba(0,0,0,0.5)';
            ctx.fillText(this.type === 'A' ? 'S' : (this.type === 'M' ? 'M' : 'P'), this.x, this.y + 4);

            ctx.restore();
        }

        applyPoison() {
            this.poisonDOT = 1; this.poisonTimer = this.poisonDuration;
        }

        getSwordTip() { return { x: this.x + Math.cos(this.swordAngle) * this.swordLength, y: this.y + Math.sin(this.swordAngle) * this.swordLength }; }
    }

    const swordTrail = [];
    let ballA, ballM, ballB;

    function init() {
        ballA = new Ball(150, 200, 22, '#ff6b6b', 'A');
        ballM = new Ball(450, 300, 20, '#8b5cf6', 'M');
        ballB = new Ball(W - 150, H - 200, 22, '#50fa7b', 'B');
        ballA.vx = 1.5; ballA.vy = 1;
        ballM.vx = -0.5; ballM.vy = -0.5;
        ballB.vx = -1.5; ballB.vy = -1;
        poisonPools.length = 0; particles.length = 0; damageNumbers.length = 0; projectiles.length = 0; explosions.length = 0; swordTrail.length = 0;
        gameState = 'countdown'; gameTime = 0;
        totalDamage = { A: 0, M: 0, B: 0 };
        screenShake = { x: 0, y: 0, intensity: 0 };
        if (gameEndTimeout) { clearTimeout(gameEndTimeout); gameEndTimeout = null; }
        document.getElementById('winner').style.display = 'none';
        document.getElementById('restartBtn').style.display = 'none';
        document.getElementById('healthSword').style.width = '100%';
        document.getElementById('healthMage').style.width = '100%';
        document.getElementById('healthPoison').style.width = '100%';
        startCountdown();
    }

    function startCountdown() {
        const countdownEl = document.getElementById('countdown');
        let count = 3;
        function showCount() {
            countdownEl.style.display = 'block'; countdownEl.textContent = count;
            countdownEl.style.animation = 'none'; countdownEl.offsetHeight; countdownEl.style.animation = 'countPulse 1s ease-out';
            if (count > 0) { count--; setTimeout(showCount, 1000); }
            else { countdownEl.textContent = 'FIGHT!'; countdownEl.style.color = '#f1fa8c'; setTimeout(() => { countdownEl.style.display = 'none'; countdownEl.style.color = '#fff'; gameState = 'playing'; }, 800); }
        }
        showCount();
    }

    function createPoisonPool(x, y) {
        x = Math.max(WALL + 15, Math.min(W - WALL - 15, x)); y = Math.max(WALL + 15, Math.min(H - WALL - 15, y));
        for (const pool of poisonPools) { const dx = pool.x - x, dy = pool.y - y; if (dx * dx + dy * dy < 400) return; }
        poisonPools.push({ x, y, radius: 30, damage: 5, pulse: Math.random() * Math.PI * 2, age: 0 });
        document.getElementById('poolCount').textContent = `POOLS: ${poisonPools.length}`;
    }

    function createPoisonCloud(x, y) {
        x = Math.max(WALL + 10, Math.min(W - WALL - 10, x)); y = Math.max(WALL + 10, Math.min(H - WALL - 10, y));
        // Create a poison cloud that persists but acts as a burst effect
        poisonPools.push({ x, y, radius: 35, damage: 6, pulse: Math.random() * Math.PI * 2, age: 0, isCloud: true });
        spawnParticles(x, y, '#50fa7b', 12);
        document.getElementById('poolCount').textContent = `POOLS: ${poisonPools.length}`;
    }

    function createExplosion(x, y, radius, damage, owner) {
        explosions.push({ x, y, radius, damage, owner, alpha: 1, scale: 0 });
        spawnParticles(x, y, '#00d9ff', 20);
        spawnParticles(x, y, '#8b5cf6', 15);
        triggerScreenShake(6);
    }

    function spawnParticles(x, y, color, count, angle = null) {
        for (let i = 0; i < count; i++) {
            const a = angle !== null ? angle + (Math.random() - 0.5) * 1.5 : Math.random() * Math.PI * 2;
            const spd = 2 + Math.random() * 4;
            particles.push({ x, y, vx: Math.cos(a) * spd, vy: Math.sin(a) * spd, life: 1, decay: 0.02 + Math.random() * 0.03, size: 2 + Math.random() * 4, color });
        }
    }

    function spawnDamageNumber(x, y, damage) {
        damageNumbers.push({ x: x + (Math.random() - 0.5) * 20, y, damage, vy: -2.5, life: 1, scale: 1 + Math.random() * 0.3 });
    }

    function triggerScreenShake(intensity) { screenShake.intensity = Math.min(intensity, 10); }

    function circleCollision(a, b) {
        if (a.hp <= -999 || b.hp <= -999) return false;
        const dx = b.x - a.x, dy = b.y - a.y; return Math.sqrt(dx * dx + dy * dy) < a.radius + b.radius;
    }
    function pointInCircle(px, py, cx, cy, r) { const dx = px - cx, dy = py - cy; return dx * dx + dy * dy < r * r; }

    function resolveCollision(a, b) {
        const dx = b.x - a.x, dy = b.y - a.y, dist = Math.sqrt(dx * dx + dy * dy);
        if (dist === 0) return;
        const nx = dx / dist, ny = dy / dist;
        const dvx = a.vx - b.vx, dvy = a.vy - b.vy, dvn = dvx * nx + dvy * ny;
        if (dvn < 0) return;
        const ma = a.radius, mb = b.radius, massSum = ma + mb;
        const impulse = (2 * dvn) / massSum;
        a.vx -= impulse * mb * nx * 0.9; a.vy -= impulse * mb * ny * 0.9;
        b.vx += impulse * ma * nx * 0.9; b.vy += impulse * ma * ny * 0.9;
        const overlap = (a.radius + b.radius - dist) / 2;
        a.x -= overlap * nx; a.y -= overlap * ny; b.x += overlap * nx; b.y += overlap * ny;
    }

    function updateProjectiles(dt, balls) {
        for (let i = projectiles.length - 1; i >= 0; i--) {
            const p = projectiles[i];
            p.trail.push({ x: p.x, y: p.y, alpha: 1 });
            if (p.trail.length > 10) p.trail.shift();
            for (const t of p.trail) t.alpha -= 0.1;

            p.x += p.vx; p.y += p.vy;

            if (p.x < WALL || p.x > W - WALL || p.y < WALL || p.y > H - WALL) {
                createExplosion(p.x, p.y, 40, p.damage, p.owner);
                projectiles.splice(i, 1); continue;
            }

            let hit = false;
            for (const ball of balls) {
                if (ball.hp <= -999) continue;
                if (ball === ballA && p.owner === 'A') continue;
                if (ball === ballM && p.owner === 'M') continue;
                if (ball === ballB && p.owner === 'B') continue;

                const dx = ball.x - p.x, dy = ball.y - p.y;
                if (dx * dx + dy * dy < (ball.radius + p.radius) ** 2) {
                    ball.takeDamage(p.damage, p.x, p.y);
                    if (p.owner === 'M') {
                        ball.applyFreeze(ball.freezeDuration);
                        totalDamage.M += p.damage;
                    } else if (p.owner === 'A') totalDamage.A += p.damage;
                    else totalDamage.B += p.damage;
                    createExplosion(p.x, p.y, 50, p.damage, p.owner);
                    hit = true; break;
                }
            }
            if (hit) { projectiles.splice(i, 1); continue; }
        }
    }

    function drawProjectiles() {
        for (const p of projectiles) {
            for (const t of p.trail) {
                if (t.alpha > 0) {
                    ctx.beginPath(); ctx.arc(t.x, t.y, 5 * t.alpha, 0, Math.PI * 2);
                    ctx.fillStyle = `rgba(139, 92, 246, ${t.alpha * 0.6})`; ctx.fill();
                }
            }
            ctx.beginPath(); ctx.arc(p.x, p.y, p.radius, 0, Math.PI * 2);
            const grad = ctx.createRadialGradient(p.x, p.y, 0, p.x, p.y, p.radius);
            grad.addColorStop(0, '#fff'); grad.addColorStop(0.3, '#8b5cf6'); grad.addColorStop(1, '#4c1d95');
            ctx.fillStyle = grad; ctx.shadowColor = '#8b5cf6'; ctx.shadowBlur = 15; ctx.fill(); ctx.shadowBlur = 0;
        }
    }

    function updateExplosions(dt) {
        for (let i = explosions.length - 1; i >= 0; i--) {
            const e = explosions[i];
            e.scale += dt * 8; e.alpha -= dt * 2;
            if (e.alpha <= 0) explosions.splice(i, 1);
        }
    }

    function drawExplosions() {
        for (const e of explosions) {
            ctx.save(); ctx.translate(e.x, e.y); ctx.scale(e.scale, e.scale);
            ctx.beginPath(); ctx.arc(0, 0, e.radius, 0, Math.PI * 2);
            ctx.strokeStyle = `rgba(0, 217, 255, ${e.alpha})`; ctx.lineWidth = 4; ctx.stroke();
            ctx.beginPath(); ctx.arc(0, 0, e.radius * 0.6, 0, Math.PI * 2);
            ctx.fillStyle = `rgba(139, 92, 246, ${e.alpha * 0.5})`; ctx.fill();
            ctx.restore();
        }
    }

    function checkSwordHit() {
        if (ballA.isSticking && ballA.stickTarget) {
            if (ballA.stickTarget.hp <= -999) {
                ballA.isSticking = false; ballA.stickTarget = null; ballA.stickHits = 0;
                return;
            }
            ballA.stickTimer -= 1/60;
            if (ballA.stickTimer <= 0) {
                ballA.stickHits++;
                ballA.stickTarget.takeDamage(8, ballA.x, ballA.y);
                totalDamage.A += 8;
                ballA.stickTimer = 0.6;
                spawnParticles(ballA.stickTarget.x, ballA.stickTarget.y, '#ff6b6b', 12); triggerScreenShake(4);
                if (ballA.stickHits >= 4) { ballA.isSticking = false; ballA.stickTarget = null; ballA.stickHits = 0; }
            }
            return;
        }

        const tip = ballA.getSwordTip();
        for (const target of [ballM, ballB]) {
            if (target.hp <= -999) continue;
            if (pointInCircle(tip.x, tip.y, target.x, target.y, target.radius)) {
                const slashAngle = ballA.swordAngle - 0.3;
                target.vx += Math.cos(slashAngle) * 5; target.vy += Math.sin(slashAngle) * 5;
                ballA.isSticking = true; ballA.stickTarget = target; ballA.stickHits = 1; ballA.stickTimer = 0.6;
                target.takeDamage(8, ballA.x, ballA.y); totalDamage.A += 8;
                spawnParticles(target.x, target.y, '#ff6b6b', 15); triggerScreenShake(6);
                break;
            }
        }
    }

    function checkPoisonTouch() {
        // Poison from ballB to ballA
        if (ballB.hp > -999 && ballA.hp > -999 && ballA.poisonDOT <= 0) {
            const dx = ballA.x - ballB.x, dy = ballA.y - ballB.y;
            if (Math.sqrt(dx * dx + dy * dy) < ballA.radius + ballB.radius + 5) {
                ballA.applyPoison(); spawnParticles(ballA.x, ballA.y, '#bd93f9', 10);
                const angle = Math.atan2(dy, dx);
                ballA.vx += Math.cos(angle) * 3; ballA.vy += Math.sin(angle) * 3;
                ballB.vx -= Math.cos(angle) * 3; ballB.vy -= Math.sin(angle) * 3;
            }
        }

        // Poison from ballM to ballA (when touching)
        if (ballM.hp > -999 && ballA.hp > -999 && ballA.poisonDOT <= 0) {
            const dx = ballA.x - ballM.x, dy = ballA.y - ballM.y;
            if (Math.sqrt(dx * dx + dy * dy) < ballA.radius + ballM.radius) {
                ballA.applyPoison(); spawnParticles(ballA.x, ballA.y, '#bd93f9', 8);
            }
        }
    }

    function checkMageCollisions() {
        // Mage touching Sword - push apart
        if (ballM.hp > -999 && ballA.hp > -999) {
            const mdx = ballM.x - ballA.x, mdy = ballM.y - ballA.y;
            if (Math.sqrt(mdx * mdx + mdy * mdy) < ballM.radius + ballA.radius) {
                const angle = Math.atan2(mdy, mdx);
                ballM.vx += Math.cos(angle) * 2; ballM.vy += Math.sin(angle) * 2;
                ballA.vx -= Math.cos(angle) * 2; ballA.vy -= Math.sin(angle) * 2;
            }
        }

        // Mage touching Poison
        if (ballM.hp > -999 && ballB.hp > -999) {
            const mdx2 = ballM.x - ballB.x, mdy2 = ballM.y - ballB.y;
            if (Math.sqrt(mdx2 * mdx2 + mdy2 * mdy2) < ballM.radius + ballB.radius) {
                const angle = Math.atan2(mdy2, mdx2);
                ballM.vx += Math.cos(angle) * 2; ballM.vy += Math.sin(angle) * 2;
                ballB.vx -= Math.cos(angle) * 2; ballB.vy -= Math.sin(angle) * 2;
            }
        }
    }

    function checkBallCollisions() {
        if (circleCollision(ballA, ballM)) {
            resolveCollision(ballA, ballM);
            if (ballA.hp > -999 && ballM.hp > -999) {
                // Create poison cloud at collision point
                const cx = (ballA.x + ballM.x) / 2, cy = (ballA.y + ballM.y) / 2;
                createPoisonCloud(cx, cy);
            }
        }
        if (circleCollision(ballA, ballB)) {
            resolveCollision(ballA, ballB);
            if (ballA.hp > -999 && ballB.hp > -999) {
                const cx = (ballA.x + ballB.x) / 2, cy = (ballA.y + ballB.y) / 2;
                createPoisonCloud(cx, cy);
            }
        }
        if (circleCollision(ballM, ballB)) {
            resolveCollision(ballM, ballB);
            if (ballM.hp > -999 && ballB.hp > -999) {
                const cx = (ballM.x + ballB.x) / 2, cy = (ballM.y + ballB.y) / 2;
                createPoisonCloud(cx, cy);
            }
        }
    }

    function checkPoisonPools() {
        for (const pool of poisonPools) {
            pool.age += 1/60; pool.pulse += 0.06;
            for (const ball of [ballA, ballM, ballB]) {
                if (ball.hp <= -999) continue;
                if (ball.type === 'B') continue; // Poison ball is immune to poison pools
                const dx = ball.x - pool.x, dy = ball.y - pool.y;
                if (dx * dx + dy * dy < (pool.radius + ball.radius) ** 2) {
                    ball.hp -= pool.damage / 60;
                    if (Math.random() < 0.02) { spawnDamageNumber(ball.x, ball.y - ball.radius - 10, Math.round(pool.damage)); spawnParticles(ball.x, ball.y, '#bd93f9', 3); }
                    if (Math.random() < 0.12 && ball.poisonDOT <= 0) { ball.poisonDOT = 1; ball.poisonTimer = 3; ball.poisonAura = 0.5; }
                }
            }
        }
    }

    function checkWinner() {
        const alive = [ballA, ballM, ballB].filter(b => b.hp > -999);

        if (alive.length === 1) {
            gameState = 'waiting';
            gameEndTimeout = setTimeout(() => {
                const winner = alive[0].type;
                endGame(winner);
            }, 500);
        } else if (alive.length === 0) {
            gameState = 'ended';
            const winnerEl = document.getElementById('winner'), textEl = document.getElementById('winnerText');
            winnerEl.style.display = 'block'; document.getElementById('restartBtn').style.display = 'block';
            winnerEl.className = 'draw';
            textEl.textContent = 'DRAW!';
        }
    }

    function updateParticles(dt) {
        for (let i = particles.length - 1; i >= 0; i--) {
            const p = particles[i];
            p.x += p.vx; p.y += p.vy; p.vy += 0.1; p.vx *= 0.98; p.vy *= 0.98; p.life -= p.decay;
            if (p.life <= 0) particles.splice(i, 1);
        }
    }

    function updateBgParticles() {
        for (const p of bgParticles) {
            p.x += p.vx; p.y += p.vy;
            if (p.x < 0) p.x = W; if (p.x > W) p.x = 0; if (p.y < 0) p.y = H; if (p.y > H) p.y = 0;
        }
    }

    function updateScreenShake(dt) {
        if (screenShake.intensity > 0) {
            screenShake.x = (Math.random() - 0.5) * screenShake.intensity * 2; screenShake.y = (Math.random() - 0.5) * screenShake.intensity * 2;
            screenShake.intensity *= 0.9;
            if (screenShake.intensity < 0.5) screenShake.intensity = 0;
        } else screenShake.x = 0; screenShake.y = 0;
    }

    function drawBackground() {
        ctx.strokeStyle = 'rgba(255,255,255,0.025)'; ctx.lineWidth = 1;
        for (let x = WALL; x < W - WALL; x += 50) { ctx.beginPath(); ctx.moveTo(x, WALL); ctx.lineTo(x, H - WALL); ctx.stroke(); }
        for (let y = WALL; y < H - WALL; y += 50) { ctx.beginPath(); ctx.moveTo(WALL, y); ctx.lineTo(W - WALL, y); ctx.stroke(); }
        const wallGrad = ctx.createLinearGradient(0, 0, W, 0);
        wallGrad.addColorStop(0, '#ff00aa'); wallGrad.addColorStop(0.5, '#00ffaa'); wallGrad.addColorStop(1, '#ff00aa');
        ctx.fillStyle = wallGrad; ctx.fillRect(0, 0, W, WALL); ctx.fillRect(0, H - WALL, W, WALL); ctx.fillRect(0, 0, WALL, H); ctx.fillRect(W - WALL, 0, WALL, H);
        ctx.strokeStyle = 'rgba(255,255,255,0.1)'; ctx.lineWidth = 2; ctx.strokeRect(WALL, WALL, W - WALL * 2, H - WALL * 2);
    }

    function drawBgParticles() { for (const p of bgParticles) { ctx.beginPath(); ctx.arc(p.x, p.y, p.size, 0, Math.PI * 2); ctx.fillStyle = `rgba(255,255,255,${p.alpha})`; ctx.fill(); } }
    function drawSwordTrail() {
        for (let i = 0; i < swordTrail.length; i++) {
            const t = swordTrail[i]; t.alpha -= 0.08;
            if (t.alpha > 0) { ctx.beginPath(); ctx.arc(t.x, t.y, 4 * t.alpha, 0, Math.PI * 2); ctx.fillStyle = `rgba(255,107,107,${t.alpha * 0.6})`; ctx.fill(); }
        }
    }

    function drawPoisonPools() {
        for (const pool of poisonPools) {
            const alpha = 0.3 + Math.sin(pool.pulse) * 0.1;
            const glow = ctx.createRadialGradient(pool.x, pool.y, 0, pool.x, pool.y, pool.radius * 2);
            glow.addColorStop(0, `rgba(189,147,249,${alpha})`); glow.addColorStop(0.4, `rgba(189,147,249,${alpha * 0.5})`); glow.addColorStop(1, 'transparent');
            ctx.beginPath(); ctx.arc(pool.x, pool.y, pool.radius * 2, 0, Math.PI * 2); ctx.fillStyle = glow; ctx.fill();
            ctx.beginPath(); ctx.arc(pool.x, pool.y, pool.radius, 0, Math.PI * 2); ctx.fillStyle = `rgba(139,92,246,${alpha + 0.1})`; ctx.fill();
            ctx.beginPath(); ctx.arc(pool.x + Math.sin(pool.pulse * 1.5) * 5, pool.y - 3, pool.radius * 0.4, 0, Math.PI * 2); ctx.fillStyle = `rgba(189,147,249,${alpha + 0.3})`; ctx.fill();
            ctx.beginPath(); ctx.arc(pool.x, pool.y, pool.radius, 0, Math.PI * 2); ctx.strokeStyle = `rgba(189,147,249,${alpha + 0.2})`; ctx.lineWidth = 2; ctx.stroke();
        }
    }

    function drawParticles() { for (const p of particles) { ctx.beginPath(); ctx.arc(p.x, p.y, p.size * p.life, 0, Math.PI * 2); ctx.fillStyle = p.color + Math.floor(p.life * 255).toString(16).padStart(2, '0'); ctx.fill(); } }

    function drawDamageNumbers() {
        ctx.font = 'bold 16px sans-serif'; ctx.textAlign = 'center';
        for (let i = damageNumbers.length - 1; i >= 0; i--) {
            const dn = damageNumbers[i];
            dn.y += dn.vy; dn.vy *= 0.95; dn.life -= 0.018;
            if (dn.life <= 0) { damageNumbers.splice(i, 1); continue; }
            ctx.fillStyle = `rgba(241,250,140,${dn.life})`;
            ctx.save(); ctx.translate(dn.x, dn.y); ctx.scale(dn.scale, dn.scale);
            ctx.fillText(typeof dn.damage === 'string' ? dn.damage : `-${dn.damage}`, 0, 0); ctx.restore();
        }
    }

    function updateHUD() {
        document.getElementById('healthSword').style.width = Math.max(0, Math.min(100, ballA.hp / ballA.maxHp * 100)) + '%';
        document.getElementById('healthMage').style.width = Math.max(0, Math.min(100, ballM.hp / ballM.maxHp * 100)) + '%';
        document.getElementById('healthPoison').style.width = Math.max(0, Math.min(100, ballB.hp / ballB.maxHp * 100)) + '%';
        document.getElementById('hpSword').textContent = ballA.hp > -999 ? Math.max(0, Math.round(ballA.hp)) : 'DEAD';
        document.getElementById('hpMage').textContent = ballM.hp > -999 ? Math.max(0, Math.round(ballM.hp)) : 'DEAD';
        document.getElementById('hpPoison').textContent = ballB.hp > -999 ? Math.max(0, Math.round(ballB.hp)) : 'DEAD';
        const mins = Math.floor(gameTime / 60), secs = Math.floor(gameTime % 60);
        document.getElementById('timer').textContent = String(mins).padStart(2, '0') + ':' + String(secs).padStart(2, '0');
        document.getElementById('damageStats').textContent = `A:${totalDamage.A} M:${totalDamage.M} B:${totalDamage.B}`;
    }

    function endGame(winner) {
        gameState = 'ended';
        const winnerEl = document.getElementById('winner'), textEl = document.getElementById('winnerText');
        winnerEl.style.display = 'block'; document.getElementById('restartBtn').style.display = 'block';
        winnerEl.className = winner === 'A' ? 'sword-win' : (winner === 'M' ? 'mage-win' : 'poison-win');
        textEl.textContent = (winner === 'A' ? 'SWORD' : (winner === 'M' ? 'MAGE' : 'POISON')) + ' WINS!';
    }

    function restart() { init(); }
    document.getElementById('restartBtn').addEventListener('click', restart);

    const balls = () => [ballA, ballM, ballB];

    function gameLoop(timestamp) {
        const dt = Math.min((timestamp - lastTime) / 1000, 0.05);
        lastTime = timestamp;

        if (gameState === 'playing' || gameState === 'waiting') {
            if (gameState === 'playing') gameTime += dt;
            ballA.update(dt, balls().filter(b => b !== ballA));
            ballM.update(dt, balls().filter(b => b !== ballM));
            ballB.update(dt, balls().filter(b => b !== ballB));

            checkBallCollisions();
            checkSwordHit();
            checkPoisonTouch();
            checkMageCollisions();
            checkPoisonPools();
            updateProjectiles(dt, balls());
            updateExplosions(dt);
            updateHUD();
        }

        updateParticles(dt); updateBgParticles(); updateScreenShake(dt);

        ctx.fillStyle = '#0a0a0f'; ctx.fillRect(0, 0, W, H);
        ctx.save();
        if (screenShake.intensity > 0) ctx.translate(screenShake.x, screenShake.y);
        drawBackground(); drawBgParticles(); drawPoisonPools(); drawSwordTrail();
        balls().forEach(b => b.draw());
        drawProjectiles(); drawExplosions(); drawParticles(); drawDamageNumbers();
        ctx.restore();

        requestAnimationFrame(gameLoop);
    }

    init();
    requestAnimationFrame(gameLoop);

    const canvas = document.getElementById('gameCanvas');

        bgParticles.push({

            vx: (Math.random() - 0.5) * 0.3, vy: (Math.random() - 0.5) * 0.3,

    class Ball {
        constructor(x, y, radius, color, type) {
            this.x = x; this.y = y; this.radius = radius;
            this.color = color; this.type = type;
            this.vx = 0; this.vy = 0;
            this.hp = 100; this.maxHp = 100;
            this.speed = 2.5; this.friction = 0.992;
            this.hitFlash = 0; this.invincible = 0;

            // Sword
            this.swordAngle = 0; this.swordLength = radius * 3.2;
            this.swordWidth = 7; this.swordRotSpeed = 3;
            this.isSticking = false; this.stickTarget = null;
            this.stickHits = 0; this.stickTimer = 0;

            // Poison
            this.poisonDOT = 0; this.poisonTimer = 0;
            this.poisonDuration = 5; this.poisonDamage = 6; // Buffed from 4 to 6
            this.poisonAura = 0; this.auraPhase = 0;

            // Mage
            this.fireCooldown = 0; this.fireRate = 1.8; // Slower firing
            this.projectileSpeed = 4; this.projectileDamage = 10; // Less damage
            this.freezeDuration = 0.8; // Shortened from 2s to 0.8s
            this.freezeImmunity = 0; // Can't be frozen again during immunity
            this.isFrozen = 0;
            this.mageAngle = 0;

            // AI
            this.aiTimer = 0; this.stuckTimer = 0;
            this.lastPos = { x, y };
        }

        update(dt, others) {
            if (this.isFrozen > 0) {
                this.isFrozen -= dt;
                this.freezeImmunity = 0.5; // Immunity lasts 0.5s after thaw
                this.vx *= 0.85; this.vy *= 0.85;
                if (this.isFrozen <= 0) { this.isFrozen = 0; }
            } else {
                this.freezeImmunity = Math.max(0, this.freezeImmunity - dt);
                this.invincible = Math.max(0, this.invincible - dt);
                this.auraPhase += dt * 3;

                if (gameState === 'playing') this.updateAI(others, dt);

                this.vx *= this.friction;
                this.vy *= this.friction;

                const spd = Math.sqrt(this.vx * this.vx + this.vy * this.vy);
                if (spd > this.speed) { this.vx = (this.vx / spd) * this.speed; this.vy = (this.vy / spd) * this.speed; }
                this.x += this.vx; this.y += this.vy;
            }

            // Wall collision (no poison on wall for poison ball - only on collision with other balls)
            if (this.x - this.radius < WALL) { this.x = WALL + this.radius; this.vx *= -0.95; }
            if (this.x + this.radius > W - WALL) { this.x = W - WALL - this.radius; this.vx *= -0.95; }
            if (this.y - this.radius < WALL) { this.y = WALL + this.radius; this.vy *= -0.95; }
            if (this.y + this.radius > H - WALL) { this.y = H - WALL - this.radius; this.vy *= -0.95; }

            // Sword rotation
            if (this.type === 'A') {
                this.swordAngle += this.swordRotSpeed * dt;
                const tipX = this.x + Math.cos(this.swordAngle) * this.swordLength;
                const tipY = this.y + Math.sin(this.swordAngle) * this.swordLength;
                swordTrail.push({ x: tipX, y: tipY, alpha: 1, color: '#ff6b6b' });
                if (swordTrail.length > 12) swordTrail.shift();
            }

            // Mage aim
            if (this.type === 'M') {
                const nearest = this.findNearest(others);
                if (nearest) {
                    const dx = nearest.x - this.x, dy = nearest.y - this.y;
                    this.mageAngle = Math.atan2(dy, dx);
                }
                this.fireCooldown -= dt;
                if (this.fireCooldown <= 0 && gameState === 'playing' && !this.isFrozen) {
                    this.fireProjectile(others);
                    this.fireCooldown = this.fireRate;
                }
            }

            if (this.hitFlash > 0) this.hitFlash -= dt * 4;

            if (this.poisonDOT > 0) {
                this.hp -= this.poisonDamage * dt;
                this.poisonTimer -= dt;
                this.poisonAura = Math.sin(this.auraPhase) * 0.3 + 0.7;
                if (this.poisonTimer <= 0) { this.poisonDOT = 0; this.poisonAura = 0; }
            }

            // Death check - mark as dead but don't end game immediately
            if (this.hp <= 0 && this.hp > -999) {
                this.hp = -1000; // Mark as dead
                spawnParticles(this.x, this.y, this.color, 30);
                if (this.type === 'B') createPoisonPool(this.x, this.y);
                checkWinner();
            }
        }

        findNearest(others) {
            let nearest = null, minDist = Infinity;
            for (const o of others) {
                if (o === this || o.hp <= 0) continue;
                const dx = o.x - this.x, dy = o.y - this.y;
                const dist = Math.sqrt(dx * dx + dy * dy);
                if (dist < minDist) { minDist = dist; nearest = o; }
            }
            return nearest;
        }

        updateAI(others, dt) {
            this.aiTimer -= dt;
            if (this.aiTimer > 0) return;
            this.aiTimer = 0.05 + Math.random() * 0.1;

            const nearest = this.findNearest(others);
            if (!nearest) return;

            const dx = nearest.x - this.x, dy = nearest.y - this.y;
            const dist = Math.sqrt(dx * dx + dy * dy);

            const moved = Math.abs(this.x - this.lastPos.x) + Math.abs(this.y - this.lastPos.y);
            if (moved < 0.5) { this.stuckTimer += dt; if (this.stuckTimer > 0.5) { this.vx += (Math.random() - 0.5) * 3; this.vy += (Math.random() - 0.5) * 3; this.stuckTimer = 0; } }
            else this.stuckTimer = 0;
            this.lastPos.x = this.x; this.lastPos.y = this.y;

            let avoidX = 0, avoidY = 0;
            // Only non-poison balls avoid pools
            if (this.type !== 'B') {
                for (const pool of poisonPools) {
                    const pdx = this.x - pool.x, pdy = this.y - pool.y;
                    const pdist = Math.sqrt(pdx * pdx + pdy * pdy);
                    if (pdist < pool.radius * 2.5) { avoidX += (pdx / pdist) * 3; avoidY += (pdy / pdist) * 3; }
                }
            }

            if (this.type === 'A') {
                const idealDist = this.swordLength * 1.2;
                if (dist > idealDist) { const a = Math.atan2(dy, dx); this.vx += Math.cos(a) * 0.5; this.vy += Math.sin(a) * 0.5; }
                else if (dist < idealDist * 0.7) { const a = Math.atan2(dy, dx); this.vx -= Math.cos(a) * 0.3; this.vy -= Math.sin(a) * 0.3; }
                else { const a = Math.atan2(dy, dx) + Math.PI / 2; this.vx += Math.cos(a) * 0.35 * (Math.random() > 0.5 ? 1 : -1); this.vy += Math.sin(a) * 0.35 * (Math.random() > 0.5 ? 1 : -1); }
            } else if (this.type === 'M') {
                const idealDist = 180;
                if (dist < idealDist * 0.6) { const a = Math.atan2(-dy, -dx); this.vx += Math.cos(a) * 0.6; this.vy += Math.sin(a) * 0.6; }
                else if (dist > idealDist * 1.8) { const a = Math.atan2(dy, dx); this.vx += Math.cos(a) * 0.4; this.vy += Math.sin(a) * 0.4; }
                else { const a = Math.atan2(dy, dx) + Math.PI / 2.5 * (Math.random() > 0.5 ? 1 : -1); this.vx += Math.cos(a) * 0.25; this.vy += Math.sin(a) * 0.25; }
            } else {
                // Poison ball - AGGRESSIVE (immune to pools, creates clouds on collision)
                if (this.poisonDOT > 0) { const a = Math.atan2(-dy, -dx); this.vx += Math.cos(a) * 0.6; this.vy += Math.sin(a) * 0.6; }
                else if (dist > this.radius * 3) {
                    // Chase aggressively when far
                    const a = Math.atan2(dy, dx); this.vx += Math.cos(a) * 0.6; this.vy += Math.sin(a) * 0.6;
                } else {
                    // Circle and deal damage when close
                    const a = Math.atan2(dy, dx) + Math.PI / 3; this.vx += Math.cos(a) * 0.5; this.vy += Math.sin(a) * 0.5;
                }

                // Poison ball is immune to pools, gains speed boost
                if (this.speed < 3.0) this.speed = Math.min(3.0, this.speed + 0.01);
            }

            this.vx += avoidX * 0.4; this.vy += avoidY * 0.4;
        }

        fireProjectile(others) {
            const nearest = this.findNearest(others);
            if (!nearest) return;

            const dx = nearest.x - this.x, dy = nearest.y - this.y;
            const angle = Math.atan2(dy, dx);

            projectiles.push({
                x: this.x + Math.cos(angle) * (this.radius + 10),
                y: this.y + Math.sin(angle) * (this.radius + 10),
                vx: Math.cos(angle) * this.projectileSpeed,
                vy: Math.sin(angle) * this.projectileSpeed,
                radius: 8,
                damage: this.projectileDamage,
                owner: this.type,
                trail: []
            });
            spawnParticles(
                this.x + Math.cos(angle) * (this.radius + 10),
                this.y + Math.sin(angle) * (this.radius + 10),
                '#8b5cf6', 6
            );
        }

        takeDamage(amount, fromX, fromY) {
            if (this.invincible > 0) return;
            this.hp -= amount;
            this.hitFlash = 1; this.invincible = 0.05;
            spawnDamageNumber(this.x, this.y - this.radius - 15, Math.round(amount));
            const angle = Math.atan2(this.y - fromY, this.x - fromX);
            spawnParticles(this.x, this.y, this.color, 8, angle);
            triggerScreenShake(amount * 0.25);
        }

        applyFreeze(duration) {
            if (this.freezeImmunity > 0) return; // Immune!
            if (this.isFrozen > 0) return; // Already frozen

            this.isFrozen = duration;
            spawnParticles(this.x, this.y, '#00d9ff', 15);
            spawnDamageNumber(this.x, this.y - this.radius - 30, 'ICE!');
        }

        draw() {
            if (this.hp <= -999) return; // Dead

            ctx.save();

            if (this.poisonDOT > 0) {
                ctx.beginPath(); ctx.arc(this.x, this.y, this.radius + 15, 0, Math.PI * 2);
                ctx.fillStyle = `rgba(189, 147, 249, ${this.poisonAura * 0.4})`; ctx.fill();
            }

            if (this.isFrozen > 0) {
                ctx.beginPath(); ctx.arc(this.x, this.y, this.radius + 12, 0, Math.PI * 2);
                ctx.fillStyle = `rgba(0, 217, 255, ${0.4 + Math.sin(performance.now() / 80) * 0.2})`; ctx.fill();

                ctx.save(); ctx.translate(this.x, this.y);
                for (let i = 0; i < 6; i++) {
                    ctx.rotate(Math.PI / 3);
                    ctx.fillStyle = `rgba(0, 217, 255, ${0.5 + Math.sin(performance.now() / 100 + i) * 0.3})`;
                    ctx.fillRect(-2, this.radius + 5, 4, 10);
                }
                ctx.restore();
            }

            const glowSize = this.radius * (this.isSticking ? 2.5 : 1.8);
            const glow = ctx.createRadialGradient(this.x, this.y, 0, this.x, this.y, glowSize);
            glow.addColorStop(0, this.color + '50'); glow.addColorStop(0.5, this.color + '20'); glow.addColorStop(1, 'transparent');
            ctx.beginPath(); ctx.arc(this.x, this.y, glowSize, 0, Math.PI * 2); ctx.fillStyle = glow; ctx.fill();

            if (this.isSticking && this.stickTarget && this.stickTarget.hp > -999) {
                ctx.beginPath(); ctx.arc(this.stickTarget.x, this.stickTarget.y, this.stickTarget.radius + 8, 0, Math.PI * 2);
                ctx.strokeStyle = '#ff6b6b'; ctx.lineWidth = 3; ctx.setLineDash([8, 4]); ctx.stroke(); ctx.setLineDash([]);
            }

            ctx.beginPath(); ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2);
            ctx.fillStyle = this.hitFlash > 0 ? `rgba(255,255,255,${this.hitFlash})` : (this.invincible > 0 ? this.color + 'aa' : this.color);
            ctx.fill();

            ctx.beginPath(); ctx.arc(this.x - this.radius * 0.3, this.y - this.radius * 0.3, this.radius * 0.35, 0, Math.PI * 2);
            ctx.fillStyle = 'rgba(255,255,255,0.35)'; ctx.fill();

            ctx.beginPath(); ctx.arc(this.x, this.y, this.radius - 2, 0, Math.PI * 2);
            ctx.strokeStyle = 'rgba(255,255,255,0.2)'; ctx.lineWidth = 2; ctx.stroke();

            if (this.type === 'A') {
                ctx.save(); ctx.translate(this.x, this.y); ctx.rotate(this.swordAngle);
                ctx.shadowColor = '#ffffff'; ctx.shadowBlur = 15;
                ctx.fillStyle = '#e8e8e8';
                ctx.fillRect(this.radius * 0.6, -this.swordWidth / 2, this.swordLength - this.radius * 0.5, this.swordWidth);
                ctx.fillStyle = '#ffffff'; ctx.fillRect(this.radius * 0.6, -this.swordWidth / 2, this.swordLength - this.radius * 0.5, 2);
                ctx.beginPath(); ctx.moveTo(this.swordLength + 5, 0); ctx.lineTo(this.swordLength - 10, -10); ctx.lineTo(this.swordLength - 10, 10); ctx.closePath(); ctx.fillStyle = '#e8e8e8'; ctx.fill();
                ctx.fillStyle = '#ffd700'; ctx.fillRect(this.radius * 0.4, -this.swordWidth - 3, 6, this.swordWidth * 2 + 6);
                ctx.restore();
            }

            if (this.type === 'M') {
                ctx.save(); ctx.translate(this.x, this.y); ctx.rotate(this.mageAngle);
                ctx.strokeStyle = '#c4b5fd'; ctx.lineWidth = 3; ctx.shadowColor = '#8b5cf6'; ctx.shadowBlur = 10;
                ctx.beginPath(); ctx.moveTo(this.radius * 0.5, 0); ctx.lineTo(this.radius + 15, 0); ctx.stroke();
                ctx.fillStyle = '#8b5cf6'; ctx.beginPath(); ctx.arc(this.radius + 15, 0, 6, 0, Math.PI * 2); ctx.fill();
                ctx.restore();

                ctx.fillStyle = '#fff';
                ctx.beginPath(); ctx.arc(this.x - 5, this.y - 4, 4, 0, Math.PI * 2); ctx.fill();
                ctx.beginPath(); ctx.arc(this.x + 5, this.y - 4, 4, 0, Math.PI * 2); ctx.fill();
                ctx.fillStyle = '#1a1a2e';
                ctx.beginPath(); ctx.arc(this.x - 4, this.y - 4, 2, 0, Math.PI * 2); ctx.fill();
                ctx.beginPath(); ctx.arc(this.x + 6, this.y - 4, 2, 0, Math.PI * 2); ctx.fill();
            }

            if (this.type === 'B') {
                // Poison bubbles
                for (let i = 0; i < 3; i++) {
                    const bx = this.x + Math.sin(performance.now() / 200 + i * 2) * 10;
                    const by = this.y + Math.cos(performance.now() / 200 + i * 2) * 10;
                    ctx.beginPath(); ctx.arc(bx, by, 3, 0, Math.PI * 2);
                    ctx.fillStyle = `rgba(189, 147, 249, ${0.3 + Math.sin(performance.now() / 100 + i) * 0.2})`; ctx.fill();
                }
            }

            ctx.font = 'bold 10px sans-serif'; ctx.textAlign = 'center';
            ctx.fillStyle = 'rgba(0,0,0,0.5)';
            ctx.fillText(this.type === 'A' ? 'S' : (this.type === 'M' ? 'M' : 'P'), this.x, this.y + 4);

            ctx.restore();
        }

        applyPoison() {
            this.poisonDOT = 1; this.poisonTimer = this.poisonDuration;

        getSwordTip() { return { x: this.x + Math.cos(this.swordAngle) * this.swordLength, y: this.y + Math.sin(this.swordAngle) * this.swordLength }; }

    }

    function createPoisonPool(x, y) {
        x = Math.max(WALL + 15, Math.min(W - WALL - 15, x)); y = Math.max(WALL + 15, Math.min(H - WALL - 15, y));
        for (const pool of poisonPools) { const dx = pool.x - x, dy = pool.y - y; if (dx * dx + dy * dy < 400) return; }
        poisonPools.push({ x, y, radius: 30, damage: 5, pulse: Math.random() * Math.PI * 2, age: 0 });
        document.getElementById('poolCount').textContent = `POOLS: ${poisonPools.length}`;
    }

    function createPoisonCloud(x, y) {
        x = Math.max(WALL + 10, Math.min(W - WALL - 10, x)); y = Math.max(WALL + 10, Math.min(H - WALL - 10, y));
        // Create a poison cloud that persists but acts as a burst effect
        poisonPools.push({ x, y, radius: 35, damage: 6, pulse: Math.random() * Math.PI * 2, age: 0, isCloud: true });
        spawnParticles(x, y, '#50fa7b', 12);
        document.getElementById('poolCount').textContent = `POOLS: ${poisonPools.length}`;
    }

    function createExplosion(x, y, radius, damage, owner) {
        explosions.push({ x, y, radius, damage, owner, alpha: 1, scale: 0 });
        spawnParticles(x, y, '#00d9ff', 20);
        spawnParticles(x, y, '#8b5cf6', 15);
        triggerScreenShake(6);
    }

    function spawnParticles(x, y, color, count, angle = null) {
        for (let i = 0; i < count; i++) {
            const a = angle !== null ? angle + (Math.random() - 0.5) * 1.5 : Math.random() * Math.PI * 2;
            const spd = 2 + Math.random() * 4;
            particles.push({ x, y, vx: Math.cos(a) * spd, vy: Math.sin(a) * spd, life: 1, decay: 0.02 + Math.random() * 0.03, size: 2 + Math.random() * 4, color });
        }
    }

    function spawnDamageNumber(x, y, damage) {
        damageNumbers.push({ x: x + (Math.random() - 0.5) * 20, y, damage, vy: -2.5, life: 1, scale: 1 + Math.random() * 0.3 });
    }

    function triggerScreenShake(intensity) { screenShake.intensity = Math.min(intensity, 10); }

    function circleCollision(a, b) {
        if (a.hp <= -999 || b.hp <= -999) return false;
        const dx = b.x - a.x, dy = b.y - a.y; return Math.sqrt(dx * dx + dy * dy) < a.radius + b.radius;
    }
    function pointInCircle(px, py, cx, cy, r) { const dx = px - cx, dy = py - cy; return dx * dx + dy * dy < r * r; }

    function resolveCollision(a, b) {
        const dx = b.x - a.x, dy = b.y - a.y, dist = Math.sqrt(dx * dx + dy * dy);
        if (dist === 0) return;
        const nx = dx / dist, ny = dy / dist;
        const dvx = a.vx - b.vx, dvy = a.vy - b.vy, dvn = dvx * nx + dvy * ny;
        if (dvn < 0) return;
        const ma = a.radius, mb = b.radius, massSum = ma + mb;
        const impulse = (2 * dvn) / massSum;
        a.vx -= impulse * mb * nx * 0.9; a.vy -= impulse * mb * ny * 0.9;
        b.vx += impulse * ma * nx * 0.9; b.vy += impulse * ma * ny * 0.9;
        const overlap = (a.radius + b.radius - dist) / 2;
        a.x -= overlap * nx; a.y -= overlap * ny; b.x += overlap * nx; b.y += overlap * ny;
    }

    function updateProjectiles(dt, balls) {
        for (let i = projectiles.length - 1; i >= 0; i--) {
            const p = projectiles[i];
            p.trail.push({ x: p.x, y: p.y, alpha: 1 });
            if (p.trail.length > 10) p.trail.shift();
            for (const t of p.trail) t.alpha -= 0.1;

            p.x += p.vx; p.y += p.vy;

            if (p.x < WALL || p.x > W - WALL || p.y < WALL || p.y > H - WALL) {
                createExplosion(p.x, p.y, 40, p.damage, p.owner);
                projectiles.splice(i, 1); continue;
            }

            let hit = false;
            for (const ball of balls) {
                if (ball.hp <= -999) continue;
                if (ball === ballA && p.owner === 'A') continue;
                if (ball === ballM && p.owner === 'M') continue;
                if (ball === ballB && p.owner === 'B') continue;

                const dx = ball.x - p.x, dy = ball.y - p.y;
                if (dx * dx + dy * dy < (ball.radius + p.radius) ** 2) {
                    ball.takeDamage(p.damage, p.x, p.y);
                    if (p.owner === 'M') {
                        ball.applyFreeze(ball.freezeDuration);
                        totalDamage.M += p.damage;
                    } else if (p.owner === 'A') totalDamage.A += p.damage;
                    else totalDamage.B += p.damage;
                    createExplosion(p.x, p.y, 50, p.damage, p.owner);
                    hit = true; break;
                }
            }
            if (hit) { projectiles.splice(i, 1); continue; }
        }
    }

    function drawProjectiles() {
        for (const p of projectiles) {
            for (const t of p.trail) {
                if (t.alpha > 0) {
                    ctx.beginPath(); ctx.arc(t.x, t.y, 5 * t.alpha, 0, Math.PI * 2);
                    ctx.fillStyle = `rgba(139, 92, 246, ${t.alpha * 0.6})`; ctx.fill();
                }
            }
            ctx.beginPath(); ctx.arc(p.x, p.y, p.radius, 0, Math.PI * 2);
            const grad = ctx.createRadialGradient(p.x, p.y, 0, p.x, p.y, p.radius);
            grad.addColorStop(0, '#fff'); grad.addColorStop(0.3, '#8b5cf6'); grad.addColorStop(1, '#4c1d95');
            ctx.fillStyle = grad; ctx.shadowColor = '#8b5cf6'; ctx.shadowBlur = 15; ctx.fill(); ctx.shadowBlur = 0;
        }
    }

    function updateExplosions(dt) {
        for (let i = explosions.length - 1; i >= 0; i--) {
            const e = explosions[i];
            e.scale += dt * 8; e.alpha -= dt * 2;
            if (e.alpha <= 0) explosions.splice(i, 1);
        }
    }

    function drawExplosions() {
        for (const e of explosions) {
            ctx.save(); ctx.translate(e.x, e.y); ctx.scale(e.scale, e.scale);
            ctx.beginPath(); ctx.arc(0, 0, e.radius, 0, Math.PI * 2);
            ctx.strokeStyle = `rgba(0, 217, 255, ${e.alpha})`; ctx.lineWidth = 4; ctx.stroke();
            ctx.beginPath(); ctx.arc(0, 0, e.radius * 0.6, 0, Math.PI * 2);
            ctx.fillStyle = `rgba(139, 92, 246, ${e.alpha * 0.5})`; ctx.fill();
            ctx.restore();
        }
    }

    function checkSwordHit() {
        if (ballA.isSticking && ballA.stickTarget) {
            if (ballA.stickTarget.hp <= -999) {
                ballA.isSticking = false; ballA.stickTarget = null; ballA.stickHits = 0;
                return;
            }
            ballA.stickTimer -= 1/60;
            if (ballA.stickTimer <= 0) {
                ballA.stickHits++;
                ballA.stickTarget.takeDamage(8, ballA.x, ballA.y);
                totalDamage.A += 8;
                ballA.stickTimer = 0.6;
                spawnParticles(ballA.stickTarget.x, ballA.stickTarget.y, '#ff6b6b', 12); triggerScreenShake(4);
                if (ballA.stickHits >= 4) { ballA.isSticking = false; ballA.stickTarget = null; ballA.stickHits = 0; }
            }
            return;
        }

        const tip = ballA.getSwordTip();
        for (const target of [ballM, ballB]) {
            if (target.hp <= -999) continue;
            if (pointInCircle(tip.x, tip.y, target.x, target.y, target.radius)) {
                const slashAngle = ballA.swordAngle - 0.3;
                target.vx += Math.cos(slashAngle) * 5; target.vy += Math.sin(slashAngle) * 5;
                ballA.isSticking = true; ballA.stickTarget = target; ballA.stickHits = 1; ballA.stickTimer = 0.6;
                target.takeDamage(8, ballA.x, ballA.y); totalDamage.A += 8;
                spawnParticles(target.x, target.y, '#ff6b6b', 15); triggerScreenShake(6);
                break;
            }
        }
    }

    function checkPoisonTouch() {
        // Poison from ballB to ballA
        if (ballB.hp > -999 && ballA.hp > -999 && ballA.poisonDOT <= 0) {
            const dx = ballA.x - ballB.x, dy = ballA.y - ballB.y;
            if (Math.sqrt(dx * dx + dy * dy) < ballA.radius + ballB.radius + 5) {
                ballA.applyPoison(); spawnParticles(ballA.x, ballA.y, '#bd93f9', 10);
                const angle = Math.atan2(dy, dx);
                ballA.vx += Math.cos(angle) * 3; ballA.vy += Math.sin(angle) * 3;
                ballB.vx -= Math.cos(angle) * 3; ballB.vy -= Math.sin(angle) * 3;
            }
        }

        // Poison from ballM to ballA (when touching)
        if (ballM.hp > -999 && ballA.hp > -999 && ballA.poisonDOT <= 0) {
            const dx = ballA.x - ballM.x, dy = ballA.y - ballM.y;
            if (Math.sqrt(dx * dx + dy * dy) < ballA.radius + ballM.radius) {
                ballA.applyPoison(); spawnParticles(ballA.x, ballA.y, '#bd93f9', 8);
            }
        }
    }

    function checkMageCollisions() {
        // Mage touching Sword - push apart
        if (ballM.hp > -999 && ballA.hp > -999) {
            const mdx = ballM.x - ballA.x, mdy = ballM.y - ballA.y;
            if (Math.sqrt(mdx * mdx + mdy * mdy) < ballM.radius + ballA.radius) {
                const angle = Math.atan2(mdy, mdx);
                ballM.vx += Math.cos(angle) * 2; ballM.vy += Math.sin(angle) * 2;
                ballA.vx -= Math.cos(angle) * 2; ballA.vy -= Math.sin(angle) * 2;
            }
        }

        // Mage touching Poison
        if (ballM.hp > -999 && ballB.hp > -999) {
            const mdx2 = ballM.x - ballB.x, mdy2 = ballM.y - ballB.y;
            if (Math.sqrt(mdx2 * mdx2 + mdy2 * mdy2) < ballM.radius + ballB.radius) {
                const angle = Math.atan2(mdy2, mdx2);
                ballM.vx += Math.cos(angle) * 2; ballM.vy += Math.sin(angle) * 2;
                ballB.vx -= Math.cos(angle) * 2; ballB.vy -= Math.sin(angle) * 2;
            }
        }
    }

    function checkBallCollisions() {
        if (circleCollision(ballA, ballM)) {
            resolveCollision(ballA, ballM);
            if (ballA.hp > -999 && ballM.hp > -999) {
                // Create poison cloud at collision point
                const cx = (ballA.x + ballM.x) / 2, cy = (ballA.y + ballM.y) / 2;
                createPoisonCloud(cx, cy);
            }
        }
        if (circleCollision(ballA, ballB)) {
            resolveCollision(ballA, ballB);
            if (ballA.hp > -999 && ballB.hp > -999) {
                const cx = (ballA.x + ballB.x) / 2, cy = (ballA.y + ballB.y) / 2;
                createPoisonCloud(cx, cy);
            }
        }
        if (circleCollision(ballM, ballB)) {
            resolveCollision(ballM, ballB);
            if (ballM.hp > -999 && ballB.hp > -999) {
                const cx = (ballM.x + ballB.x) / 2, cy = (ballM.y + ballB.y) / 2;
                createPoisonCloud(cx, cy);
            }
        }
    }

    function checkPoisonPools() {
        for (const pool of poisonPools) {
            pool.age += 1/60; pool.pulse += 0.06;
            for (const ball of [ballA, ballM, ballB]) {
                if (ball.hp <= -999) continue;
                if (ball.type === 'B') continue; // Poison ball is immune to poison pools
                const dx = ball.x - pool.x, dy = ball.y - pool.y;
                if (dx * dx + dy * dy < (pool.radius + ball.radius) ** 2) {
                    ball.hp -= pool.damage / 60;
                    if (Math.random() < 0.02) { spawnDamageNumber(ball.x, ball.y - ball.radius - 10, Math.round(pool.damage)); spawnParticles(ball.x, ball.y, '#bd93f9', 3); }
                    if (Math.random() < 0.12 && ball.poisonDOT <= 0) { ball.poisonDOT = 1; ball.poisonTimer = 3; ball.poisonAura = 0.5; }
                }
            }
        }
    }

    function checkWinner() {
        const alive = [ballA, ballM, ballB].filter(b => b.hp > -999);

        if (alive.length === 1) {
            gameState = 'waiting';
            gameEndTimeout = setTimeout(() => {
                const winner = alive[0].type;
                endGame(winner);
            }, 500);
        } else if (alive.length === 0) {
            gameState = 'ended';
            const winnerEl = document.getElementById('winner'), textEl = document.getElementById('winnerText');
            winnerEl.style.display = 'block'; document.getElementById('restartBtn').style.display = 'block';
            winnerEl.className = 'draw';
            textEl.textContent = 'DRAW!';
        }
    }

    function updateParticles(dt) {
        for (let i = particles.length - 1; i >= 0; i--) {
            const p = particles[i];
            p.x += p.vx; p.y += p.vy; p.vy += 0.1; p.vx *= 0.98; p.vy *= 0.98; p.life -= p.decay;
            if (p.life <= 0) particles.splice(i, 1);
        }
    }

    function updateBgParticles() {
        for (const p of bgParticles) {
            p.x += p.vx; p.y += p.vy;
            if (p.x < 0) p.x = W; if (p.x > W) p.x = 0; if (p.y < 0) p.y = H; if (p.y > H) p.y = 0;
        }
    }

    function updateScreenShake(dt) {
        if (screenShake.intensity > 0) {
            screenShake.x = (Math.random() - 0.5) * screenShake.intensity * 2; screenShake.y = (Math.random() - 0.5) * screenShake.intensity * 2;
            screenShake.intensity *= 0.9;
            if (screenShake.intensity < 0.5) screenShake.intensity = 0;
        } else screenShake.x = 0; screenShake.y = 0;
    }

    function drawBackground() {
        ctx.strokeStyle = 'rgba(255,255,255,0.025)'; ctx.lineWidth = 1;
        for (let x = WALL; x < W - WALL; x += 50) { ctx.beginPath(); ctx.moveTo(x, WALL); ctx.lineTo(x, H - WALL); ctx.stroke(); }
        for (let y = WALL; y < H - WALL; y += 50) { ctx.beginPath(); ctx.moveTo(WALL, y); ctx.lineTo(W - WALL, y); ctx.stroke(); }
        const wallGrad = ctx.createLinearGradient(0, 0, W, 0);
        wallGrad.addColorStop(0, '#ff00aa'); wallGrad.addColorStop(0.5, '#00ffaa'); wallGrad.addColorStop(1, '#ff00aa');
        ctx.fillStyle = wallGrad; ctx.fillRect(0, 0, W, WALL); ctx.fillRect(0, H - WALL, W, WALL); ctx.fillRect(0, 0, WALL, H); ctx.fillRect(W - WALL, 0, WALL, H);
        ctx.strokeStyle = 'rgba(255,255,255,0.1)'; ctx.lineWidth = 2; ctx.strokeRect(WALL, WALL, W - WALL * 2, H - WALL * 2);
    }

    function drawBgParticles() { for (const p of bgParticles) { ctx.beginPath(); ctx.arc(p.x, p.y, p.size, 0, Math.PI * 2); ctx.fillStyle = `rgba(255,255,255,${p.alpha})`; ctx.fill(); } }
    function drawSwordTrail() {
        for (let i = 0; i < swordTrail.length; i++) {
            const t = swordTrail[i]; t.alpha -= 0.08;
            if (t.alpha > 0) { ctx.beginPath(); ctx.arc(t.x, t.y, 4 * t.alpha, 0, Math.PI * 2); ctx.fillStyle = `rgba(255,107,107,${t.alpha * 0.6})`; ctx.fill(); }
        }
    }

    function drawPoisonPools() {
        for (const pool of poisonPools) {
            const alpha = 0.3 + Math.sin(pool.pulse) * 0.1;
            const glow = ctx.createRadialGradient(pool.x, pool.y, 0, pool.x, pool.y, pool.radius * 2);
            glow.addColorStop(0, `rgba(189,147,249,${alpha})`); glow.addColorStop(0.4, `rgba(189,147,249,${alpha * 0.5})`); glow.addColorStop(1, 'transparent');
            ctx.beginPath(); ctx.arc(pool.x, pool.y, pool.radius * 2, 0, Math.PI * 2); ctx.fillStyle = glow; ctx.fill();
            ctx.beginPath(); ctx.arc(pool.x, pool.y, pool.radius, 0, Math.PI * 2); ctx.fillStyle = `rgba(139,92,246,${alpha + 0.1})`; ctx.fill();
            ctx.beginPath(); ctx.arc(pool.x + Math.sin(pool.pulse * 1.5) * 5, pool.y - 3, pool.radius * 0.4, 0, Math.PI * 2); ctx.fillStyle = `rgba(189,147,249,${alpha + 0.3})`; ctx.fill();
            ctx.beginPath(); ctx.arc(pool.x, pool.y, pool.radius, 0, Math.PI * 2); ctx.strokeStyle = `rgba(189,147,249,${alpha + 0.2})`; ctx.lineWidth = 2; ctx.stroke();
        }
    }

    function drawParticles() { for (const p of particles) { ctx.beginPath(); ctx.arc(p.x, p.y, p.size * p.life, 0, Math.PI * 2); ctx.fillStyle = p.color + Math.floor(p.life * 255).toString(16).padStart(2, '0'); ctx.fill(); } }

    function drawDamageNumbers() {
        ctx.font = 'bold 16px sans-serif'; ctx.textAlign = 'center';
        for (let i = damageNumbers.length - 1; i >= 0; i--) {
            const dn = damageNumbers[i];
            dn.y += dn.vy; dn.vy *= 0.95; dn.life -= 0.018;
            if (dn.life <= 0) { damageNumbers.splice(i, 1); continue; }
            ctx.fillStyle = `rgba(241,250,140,${dn.life})`;
            ctx.save(); ctx.translate(dn.x, dn.y); ctx.scale(dn.scale, dn.scale);
            ctx.fillText(typeof dn.damage === 'string' ? dn.damage : `-${dn.damage}`, 0, 0); ctx.restore();
        }
    }

    function updateHUD() {
        document.getElementById('healthSword').style.width = Math.max(0, Math.min(100, ballA.hp / ballA.maxHp * 100)) + '%';
        document.getElementById('healthMage').style.width = Math.max(0, Math.min(100, ballM.hp / ballM.maxHp * 100)) + '%';
        document.getElementById('healthPoison').style.width = Math.max(0, Math.min(100, ballB.hp / ballB.maxHp * 100)) + '%';
        document.getElementById('hpSword').textContent = ballA.hp > -999 ? Math.max(0, Math.round(ballA.hp)) : 'DEAD';
        document.getElementById('hpMage').textContent = ballM.hp > -999 ? Math.max(0, Math.round(ballM.hp)) : 'DEAD';
        document.getElementById('hpPoison').textContent = ballB.hp > -999 ? Math.max(0, Math.round(ballB.hp)) : 'DEAD';
        const mins = Math.floor(gameTime / 60), secs = Math.floor(gameTime % 60);
        document.getElementById('timer').textContent = String(mins).padStart(2, '0') + ':' + String(secs).padStart(2, '0');
        document.getElementById('damageStats').textContent = `A:${totalDamage.A} M:${totalDamage.M} B:${totalDamage.B}`;
    }

    function endGame(winner) {
        gameState = 'ended';
        const winnerEl = document.getElementById('winner'), textEl = document.getElementById('winnerText');
        winnerEl.style.display = 'block'; document.getElementById('restartBtn').style.display = 'block';
        winnerEl.className = winner === 'A' ? 'sword-win' : (winner === 'M' ? 'mage-win' : 'poison-win');
        textEl.textContent = (winner === 'A' ? 'SWORD' : (winner === 'M' ? 'MAGE' : 'POISON')) + ' WINS!';
    }

    function restart() { init(); }
    document.getElementById('restartBtn').addEventListener('click', restart);

    const balls = () => [ballA, ballM, ballB];

    function gameLoop(timestamp) {
        const dt = Math.min((timestamp - lastTime) / 1000, 0.05);
        lastTime = timestamp;

        if (gameState === 'playing' || gameState === 'waiting') {
            if (gameState === 'playing') gameTime += dt;
            ballA.update(dt, balls().filter(b => b !== ballA));
            ballM.update(dt, balls().filter(b => b !== ballM));
            ballB.update(dt, balls().filter(b => b !== ballB));

            checkBallCollisions();
            checkSwordHit();
            checkPoisonTouch();
            checkMageCollisions();
            checkPoisonPools();
            updateProjectiles(dt, balls());
            updateExplosions(dt);
            updateHUD();
        }

        updateParticles(dt); updateBgParticles(); updateScreenShake(dt);

        ctx.fillStyle = '#0a0a0f'; ctx.fillRect(0, 0, W, H);
        ctx.save();
        if (screenShake.intensity > 0) ctx.translate(screenShake.x, screenShake.y);
        drawBackground(); drawBgParticles(); drawPoisonPools(); drawSwordTrail();
        balls().forEach(b => b.draw());
        drawProjectiles(); drawExplosions(); drawParticles(); drawDamageNumbers();
        ctx.restore();

        requestAnimationFrame(gameLoop);
    }

    init();
    requestAnimationFrame(gameLoop);
